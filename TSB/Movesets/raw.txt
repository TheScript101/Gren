-- Utility Functions

local function hideBasePart(obj)

	if obj:IsA("BasePart") then

		obj.Transparency = 1

		obj.CanCollide = false

		obj.CastShadow = false

	end

end

local function setVFXRed(obj)

	if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") then

		obj.Color = ColorSequence.new(Color3.fromRGB(139, 0, 0))

	end

end

-- Function to attach model to an arm

local function attachToArm(arm, modelTemplate)

	local dClone = modelTemplate:Clone()

	if dClone:IsA("Part") then

		dClone.Parent = arm

		dClone.CFrame = arm.CFrame

		hideBasePart(dClone)

		for _, obj in ipairs(dClone:GetDescendants()) do

			hideBasePart(obj)

			setVFXRed(obj)

		end

		local weld = Instance.new("WeldConstraint")

		weld.Part0 = arm

		weld.Part1 = dClone

		weld.Parent = dClone

	elseif dClone:IsA("Model") then

		if not dClone.PrimaryPart then

			warn("Model has no PrimaryPart!")

			return

		end

		dClone.Parent = workspace

		dClone:SetPrimaryPartCFrame(

			arm.CFrame *

			CFrame.new(MoveRight - MoveLeft, MoveUp - MoveDown, MoveBackward - MoveForward) *

			CFrame.Angles(RotateRight - RotateLeft, RotateUp - RotateDown, RotateBackward - RotateForward)

		)

		for _, obj in ipairs(dClone:GetDescendants()) do

			hideBasePart(obj)

			setVFXRed(obj)

		end

		local weld = Instance.new("WeldConstraint")

		weld.Part0 = arm

		weld.Part1 = dClone.PrimaryPart

		weld.Parent = dClone.PrimaryPart

	end

end
